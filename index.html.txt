  <script>
    const input = document.getElementById("inputText");
    const output = document.getElementById("outputArea");

    const styles = {
      "Bold Text": (text) => text.replace(/./g, c => {
        const code = c.charCodeAt(0);
        return code >= 65 && code <= 90 ? String.fromCharCode(0x1D400 + code - 65) :
               code >= 97 && code <= 122 ? String.fromCharCode(0x1D41A + code - 97) : c;
      }),
      "Big Text": (text) => text.split('').map(char => {
        const index = 'ABCDEFGHIJKLMNOPQRSTUVWXYZ'.indexOf(char.toUpperCase());
        return index !== -1 ? '🅐🅑🅒🅓🅔🅕🅖🅗🅘🅙🅚🅛🅜🅝🅞🅟🅠🅡🅢🅣🅤🅥🅦🅧🅨🅩'[index] : char;
      }).join(''),
      "Cursed Text": (text) => text.split('').map(c => c + '\u0336').join(''),
      "Whitespace Remover": (text) => text.replace(/\s+/g, ''),
      "Underlined Text": (text) => text.split('').map(c => c + '\u0332').join(''),
      "Weird Text": (text) => text.replace(/A/g, '𝔄').replace(/e/g, 'є').replace(/i/g, 'ι').replace(/r/g, 'я').replace(/t/g, 'т'),
      "Wingdings Style": (text) => text.replace(/./g, c => '✪☛❂♞⚑☂♻✿'.charAt(Math.floor(Math.random() * 8)))
    };

    input.addEventListener("input", () => {
      const value = input.value;
      output.innerHTML = "";

      Object.entries(styles).forEach(([name, fn]) => {
        const converted = fn(value);
        const box = document.createElement("div");
        box.className = "output-box";
        box.innerHTML = `<span>${converted}</span> <button class='copy-btn'>Copy</button>`;

        const button = box.querySelector(".copy-btn");
        button.addEventListener("click", () => {
          navigator.clipboard.writeText(converted);
          button.textContent = "Copied!";
          setTimeout(() => button.textContent = "Copy", 2000);
        });

        output.appendChild(box);
      });
    });
  </script>
